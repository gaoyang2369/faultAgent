"""数据提取和可视化工具 -- extract_data 与 fig_inter 共享 globals() 命名空间。

IMPORTANT: extract_data 和 fig_inter 必须在同一文件中，
因为 extract_data 通过 globals()[df_name] = df 存储 DataFrame，
fig_inter 通过 globals() 读取这些 DataFrame。分到不同文件会导致命名空间隔离。
"""
import os
import matplotlib
import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd

from dotenv import load_dotenv
from pydantic import BaseModel, Field
from langchain_core.tools import tool
from sqlalchemy import create_engine


# === extract_data 工具 ===
class ExtractQuerySchema(BaseModel):
    sql_query: str = Field(description="SQL查询语句，用于从MySQL提取数据")
    df_name: str = Field(description="pandas变量名，命名规范：df_{表名}或df_{用途}")


@tool(args_schema=ExtractQuerySchema)
def extract_data(sql_query: str, df_name: str) -> str:
    """
    将机械臂数据库查询结果加载为pandas DataFrame。

    使用场景：需要用fig_inter工具绘制机械臂数据的图表时。

    使用建议：
    - 变量命名规范为`df_{表名}`或`df_{用途}`，如`df_fault_data`、`df_normal_data`
    - 适合复杂的数据分析、建模、可视化等操作

    注意：仅负责数据提取，不负责SQL查询。如需查询，请使用sql_inter工具。
    """
    load_dotenv(override=True)
    host = os.getenv('HOST')
    user = os.getenv('USER')
    mysql_pw = os.getenv('MYSQL_PW')
    db = os.getenv('DB_NAME')
    port = os.getenv('PORT')

    engine_url = f"mysql+pymysql://{user}:{mysql_pw}@{host}:{port}/{db}?charset=utf8mb4"
    engine = create_engine(engine_url, pool_pre_ping=True)

    try:
        df = pd.read_sql(sql_query, engine)
        globals()[df_name] = df
        return f"✅ 成功创建 pandas 对象 `{df_name}`，包含从 MySQL 提取的数据。"
    except Exception as e:
        return f"❌ 执行失败：{e}"
    finally:
        try:
            engine.dispose()
        except Exception:
            pass


# === fig_inter 工具（主Agent版，使用 sns/pd） ===
class FigCodeInput(BaseModel):
    py_code: str = Field(description="Python绘图代码，使用matplotlib/seaborn创建图像并赋值给变量")
    fname: str = Field(description="图对象变量名，命名规范：场景+核心指标+图表类型，如fault_current_trend_fig_20251121_1530")


@tool(args_schema=FigCodeInput)
def fig_inter(py_code: str, fname: str) -> str:
    """
    生成数据可视化图表。

    使用场景：需要可视化展示数据分析结果时。

    核心功能：生成数据可视化图表并保存为图片。

    命名规范：`fname`按「场景+核心指标+图表类型」命名，如：
    - fault_current_trend_fig（故障电流趋势图）

    代码要求：
    - 图对象变量名必须与`fname`完全一致
    - 必须使用`{fname} = plt.figure()`或`{fname}, axes = plt.subplots()`
    - 必须设置中文字体：`plt.rcParams["font.family"] = ["SimHei", "Microsoft YaHei"]`
    - 设置负号显示：`plt.rcParams["axes.unicode_minus"] = False`
    - 最后调用`{fname}.tight_layout()`
    - 禁止使用`plt.show()`

    图表类型建议：
    - 时序趋势：折线图（多曲线对比）
    - 对比分析：柱状图、箱线图
    - 分布分析：直方图、密度图
    - 关联分析：热力图、散点图
    - 异常标记：在图表中标记异常点、阈值线
    """
    current_backend = matplotlib.get_backend()
    matplotlib.use('Agg')

    local_vars = {"plt": plt, "pd": pd, "sns": sns}

    # __file__ 路径调整：tools/ 下文件需要 dirname(dirname(__file__)) 到项目根
    root_dir = os.path.join(os.path.dirname(os.path.dirname(__file__)), "agent_fronted")
    public_dir = os.path.join(root_dir, "public")
    images_dir = os.path.join(public_dir, "images")
    os.makedirs(images_dir, exist_ok=True)

    try:
        g = globals()
        exec(py_code, g, local_vars)
        g.update(local_vars)

        fig = local_vars.get(fname, None)
        if not fig:
            return "⚠️ 未找到图对象，请确保代码中变量名与fname一致"

        image_filename = f"{fname}.png"
        abs_path = os.path.join(images_dir, image_filename)
        web_path = f"/images/{image_filename}"
        report_path = f"../images/{image_filename}"

        fig.savefig(abs_path, bbox_inches='tight')
        return f"✅ 图片保存成功！前端展示路径：{web_path}，报告中引用路径：{report_path}"

    except Exception as e:
        return f"❌ 执行失败：{e}"
    finally:
        plt.close('all')
        matplotlib.use(current_backend)

"""SQL查询工具 -- 延迟初始化数据库连接。"""
import os
import json
import pymysql
import pandas as pd
from datetime import datetime
from dotenv import load_dotenv
from pydantic import BaseModel, Field
from langchain_core.tools import tool
from langchain_community.utilities import SQLDatabase
from langchain_community.agent_toolkits import SQLDatabaseToolkit
from langchain_openai import ChatOpenAI
from config import DCMA_DB_NAME

# === 延迟初始化单例 ===
_db = None
_sqltools = None


def _get_db():
    """懒加载 SQLDatabase 单例。"""
    global _db
    if _db is None:
        load_dotenv(override=True)
        host = os.getenv('HOST')
        user = os.getenv('USER')
        mysql_pw = os.getenv('MYSQL_PW')
        port = os.getenv('PORT')
        _db = SQLDatabase.from_uri(
            f"mysql+pymysql://{user}:{mysql_pw}@{host}:{port}/{DCMA_DB_NAME}"
        )
    return _db


def get_sqltools():
    """返回 SQLDatabaseToolkit 生成的工具列表。在 app lifespan 中 DB 就绪后调用。"""
    global _sqltools
    if _sqltools is None:
        db = _get_db()
        load_dotenv(override=True)
        model = ChatOpenAI(
            model=os.getenv("MODEL_NAME"),
            base_url=os.getenv("OPENAI_BASE_URL"),
            api_key=os.getenv("OPENAI_API_KEY"),
            temperature=0.7,
        )
        toolkit = SQLDatabaseToolkit(db=db, llm=model)
        _sqltools = toolkit.get_tools()
    return _sqltools


# === sql_inter 工具（每次调用新建短命连接） ===
class SQLQuerySchema(BaseModel):
    sql_query: str = Field(description="SQL查询语句，建议使用LIMIT限制返回行数")


@tool(args_schema=SQLQuerySchema)
def sql_inter(sql_query: str) -> str:
    """
    在机械臂数据库上执行SQL查询。

    使用场景：
    - 快速预览数据（建议LIMIT）
    - 查询数据库结构

    注意：仅执行查询，不进行数据分析。如需分析数据，使用extract_data。
    """
    print("正在调用 sql_inter 工具运行 SQL 查询...")

    load_dotenv(override=True)
    host = os.getenv('HOST')
    user = os.getenv('USER')
    mysql_pw = os.getenv('MYSQL_PW')
    db = os.getenv('DB_NAME')
    port = os.getenv('PORT')

    connection = pymysql.connect(
        host=host, user=user, passwd=mysql_pw, db=db, port=int(port), charset='utf8'
    )

    try:
        with connection.cursor() as cursor:
            cursor.execute(sql_query)
            results = cursor.fetchall()
            print("SQL 查询已成功执行，正在整理结果...")

            MAX_ROWS = 50
            total_rows = len(results)
            truncated = False
            preview_rows = results
            if total_rows > MAX_ROWS:
                preview_rows = results[:MAX_ROWS]
                truncated = True

        def json_serial(obj):
            if isinstance(obj, (datetime, pd.Timestamp)):
                return obj.isoformat()
            raise TypeError(f"Type {type(obj)} not serializable")

        payload = {
            "rows": preview_rows,
            "total_rows": total_rows,
            "truncated": truncated,
            "note": "已仅返回前50行预览。。" if truncated else ""
        }
        return json.dumps(payload, ensure_ascii=False, default=json_serial)
    except Exception as e:
        return f"SQL执行失败：{e}"
    finally:
        connection.close()

"""知识库检索工具。"""
from pydantic import BaseModel, Field
from langchain_core.tools import tool


class KnowledgeBaseQuerySchema(BaseModel):
    query: str = Field(description="检索查询字符串，需明确具体")


@tool(args_schema=KnowledgeBaseQuerySchema)
def query_knowledge_base(query: str) -> str:
    """
    查询本地知识库获取元信息（故障代码含义）。

    使用场景：
    - 需要查询故障代码含义、触发原因、处理步骤（如：F01002的触发原因、处理步骤）

    注意：知识库仅包含元信息，不包含实际数据。
    """
    try:
        from knowledge_base import db_retriever
        if db_retriever is None:
            return "错误：本地知识库未初始化，请先创建知识库。"
        try:
            import concurrent.futures
            timeout_seconds = 8
            with concurrent.futures.ThreadPoolExecutor(max_workers=1) as executor:
                future = executor.submit(db_retriever.invoke, query)
                relevant_docs = future.result(timeout=timeout_seconds)
        except concurrent.futures.TimeoutError:
            return f"超时：知识库检索超过 {timeout_seconds}s 未返回，请稍后重试或缩小查询范围。"
        except Exception as e:
            return f"知识库检索失败：{e}"
        return "\n\n".join([f"文档片段：{doc.page_content}\n来源页码：{doc.metadata.get('page', '未知')}"
                            for doc in relevant_docs[:3]])
    except ImportError:
        return "错误：知识库模块未找到"
