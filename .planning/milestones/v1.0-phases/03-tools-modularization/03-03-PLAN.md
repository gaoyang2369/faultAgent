**Step 1: Create tests/test_lazy_init.py** (Wave 0 Nyquist requirement -- proves TOOL-02 lazy init).

This test verifies that `from tools import tools` does NOT trigger MySQL or Ollama connections. It patches the connection entry points and asserts they were NOT called during import.

```python
"""TOOL-02 验证：导入 tools 包不触发 MySQL/Ollama 连接。

Wave 0 Nyquist test -- proves lazy initialization works.
"""
import importlib
import sys
from unittest.mock import patch, MagicMock


def test_import_tools_no_db():
    """from tools import tools 不应触发 SQLDatabase.from_uri 或 pymysql.connect。"""
    # Remove cached tools modules to test fresh import
    modules_to_remove = [k for k in sys.modules if k.startswith("tools")]
    for mod in modules_to_remove:
        del sys.modules[mod]

    with patch("langchain_community.utilities.SQLDatabase.from_uri") as mock_db_uri, \
         patch("pymysql.connect") as mock_pymysql:
        # Re-import tools package
        import tools
        importlib.reload(tools)

        # Neither should have been called during import
        mock_db_uri.assert_not_called()
        mock_pymysql.assert_not_called()

        # tools list should have 9 entries
        assert len(tools.tools) == 9, f"Expected 9 tools, got {len(tools.tools)}"


def test_get_sqltools_lazy_singleton():
    """get_sqltools() 应该是懒加载单例 -- 调用前 _db 为 None。"""
    from tools.sql_tools import _db, _sqltools
    # At module level (before any call), both should be None
    # (unless a previous test triggered them -- reset first)
    import tools.sql_tools as st
    st._db = None
    st._sqltools = None

    assert st._db is None, "_db should be None before get_sqltools() call"
    assert st._sqltools is None, "_sqltools should be None before get_sqltools() call"
```
